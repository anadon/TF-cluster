00300 inline bool AffxString::isNumber()
00301 {
00302     // empty string
00303     if (length() == 0) { return false; }
00304 
00305     bool bIsNumber = true;
00306     for (int i = 0; i < (int)length(); i++)
00307     {
00308         // must be a digit
00309         // must be a period and it must be in the middle
00310         // must be a minus sign and it must be at the beginning
00311         if (!isdigit(getAt(i)) && 
00312             !(getAt(i) == '.' && 0 < i && i < (int)length() - 1) &&
00313             !(getAt(i) == '-' && i == 0) &&
00314             !(getAt(i) == '+' && i == 0))
00315         {
00316             bIsNumber = false;
00317             break;
00318         }
00319     }
00320     return bIsNumber;
