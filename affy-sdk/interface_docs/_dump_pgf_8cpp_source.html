00537 void dumpPgf::splitTypes (const std::string& inputString, std::vector<std::string>& inputTypes)
00538 {
00539   // Want to reuse the output vector; clear it first.
00540   inputTypes.clear();
00541   size_t substrBegin = 0;
00542   for (;;)
00543   {
00544     size_t substrEnd = inputString.find ("->", substrBegin);
00545     if (substrEnd == string::npos)
00546     {
00547       // No more '->' - save what's left, quit.
00548       string subString = inputString.substr (substrBegin);
00549       // Avoid returning a null string from a terminating '->' or an empty inputString.
00550       if (! subString.empty())
00551         inputTypes.push_back (subString);
00552       break;
00553     }
00554     // Avoid null strings from an initial '->' or '->->'.
00555     if (substrEnd != substrBegin)
00556       inputTypes.push_back (inputString.substr (substrBegin, substrEnd - substrBegin) );
00557     // Continue following the '->'
00558     substrBegin = substrEnd + 2;
00559   }
