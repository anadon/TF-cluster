00919 bool Fs::isBinaryFile(const std::string& path)
00920 {
00921   std::fstream fs;
00922   
00923   Fs::aptOpen(fs, path, std::ios::in|std::ios::binary);
00924   if (!fs.good()) {
00925     return false;
00926   }
00927 
00928   int cnt_txt=0;
00929   int cnt_bin=0;
00930   for (int i=0;i<1024;i++) {
00931     int c=fs.get();
00932     // the end?
00933     if (c==-1) {
00934       break;
00935     }
00936     // text?
00937     if (((c>=0x20)&&(c<=0x7E)) // " " -- "~"
00938         || (c==0x09) // "\t"
00939         || (c==0x0A) // "\n"
00940         || (c==0x0D) // "\r"
00941         ) {
00942       cnt_txt++;
00943       continue;
00944     }
00945     cnt_bin++;
00946   }
00947 
00948   // maybe compute some ratio?
00949   double cnt_total=cnt_bin+cnt_txt;
00950   if (cnt_total==0) {
00951     return false;
00952   }
00953   // for now, if there are any, we take the file as being binary.
00954   if (cnt_bin>0) {
00955     return true;
00956   }
00957   // not binary. (Which is not the same as text.)
00958   return false;

01484               Util::chopString(sp,':',searchPathVec);
01485             }
01486         }
01487

01492     }
