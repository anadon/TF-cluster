00365 void
00366 tsv_util_paste(const std::string& out_fileName,
00367                const std::vector<std::string> in_tsv_fileNames,
00368                int seg_max_lines,
00369                const std::string& key_col_name,
00370                int verbose,
00371                int tsv_paste_fh_limit)
00372 {
00373   // the simple case, no passes needed...
00374   if (in_tsv_fileNames.size()<=tsv_paste_fh_limit) {
00375     // ...so just do it.
00376     tsv_util_paste_basic(out_fileName,
00377                          in_tsv_fileNames,in_tsv_fileNames,
00378                          seg_max_lines,key_col_name,verbose);
00379     // and we are done.
00380     return;
00381   }
00382 
00383   // complex case, passes are needed.
00384   if (verbose>=1) {
00385     printf("tsv_util_paste: multiple passes needed to process '%d' inputs.\n",
00386            int(in_tsv_fileNames.size()));
00387   }
00388 
00389   // we have to be able to open at least two files at a time.
00390   if (tsv_paste_fh_limit<=1) {
00391     Err::errAbort("tsv_paste_fh_limit must be bigger than 2.");
00392   }
00393 
00394   //
00395   std::vector<std::string> copy_fileNames=in_tsv_fileNames;
00396   //
00397   int pass_cnt=0;
00398   std::string pass0_out_name;
00399   std::string pass1_out_name;
00400   std::vector<std::string> pass_out_names;
00401   std::vector<std::string> pass1_fileNames;
00402 
00403   // while we have more than the limit, we need to use a temp file.
00404   while (copy_fileNames.size()>tsv_paste_fh_limit) {
00405     pass1_fileNames.clear();
00406     // tack on the prior tmp file.
00407     if (pass0_out_name!="") {
00408       pass1_fileNames.push_back(pass0_out_name);
00409     }
00410     // add some more input files.
00411     // dont adjust the length for the file we pushed on.
00412     int len=tsv_paste_fh_limit;
00413     pass1_fileNames.insert(pass1_fileNames.end(),
00414                            copy_fileNames.begin(),copy_fileNames.begin()+len);
00415     copy_fileNames.erase(copy_fileNames.begin(),copy_fileNames.begin()+len);
00416     // gen the name for the new temp file.
00417     char buf[100];
00418     SNPRINTF(buf,sizeof(buf),"-%05d.tmp",pass_cnt);
00419     pass1_out_name=out_fileName+buf;
00420     pass_out_names.push_back(pass1_out_name);
00421     //
00422     if (verbose>=1) {
00423       printf("pass: %d : '%s'...\n",pass_cnt,pass1_out_name.c_str());
00424     }
00425     // paste what we have into the new tmp file.
00426     tsv_util_paste_basic(pass1_out_name,
00427                          pass1_fileNames,
00428                          pass1_fileNames, // the filenames in this pass
00429                          0, // dont segment.
00430                          key_col_name,verbose);
00431     // get rid of the old tmp file.
00432     if (pass0_out_name!="") {
00433       //printf("rm %s\n",pass0_out_name.c_str());
00434       Fs::rm(pass0_out_name);
00435     }
00436     // the current is now the old.
00437     pass_cnt++;
00438     pass0_out_name=pass1_out_name;
00439   }
00440   // now do the last pass into the official output file.
00441   pass1_fileNames.clear();
00442   if (pass0_out_name!="") {
00443     pass1_fileNames.push_back(pass0_out_name);
00444   }
00445   pass1_fileNames.insert(pass1_fileNames.end(),copy_fileNames.begin(),copy_fileNames.end());
00446   tsv_util_paste_basic(out_fileName, // the final output file
00447                        pass1_fileNames,
00448                        in_tsv_fileNames, // the filenames for all the inputs.
00449                        seg_max_lines, // do segment if needed.
00450                        key_col_name,verbose);
