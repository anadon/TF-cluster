00050 {
00051         fileName = file;
00052         GenericData data;
00053         GenericFileReader reader;
00054         reader.SetFilename(fileName);
00055         reader.Open(data);

00057         int ng = data.Header().GetDataGroupCnt();
00058         positions.resize(ng);
00059         rowsizes.resize(ng);
00060         colsizes.resize(ng);
00061     dataSetNames.resize(ng);
00062         for (int ig=0; ig<ng; ig++)
00063         {
00064                 DataGroupHeader &dgHdr = data.Header().GetDataGroup(ig);
00065                 int ns = dgHdr.GetDataSetCnt();
00066                 positions[ig].resize(ns);
00067                 rowsizes[ig].resize(ns);
00068                 colsizes[ig].resize(ns);
00069         dataSetNames[ig].resize(ns);
00070                 for (int is=0; is<ns; is++)
00071                 {
00072                         DataSetHeader &dsHdr = dgHdr.GetDataSet(is);
00073                         positions[ig][is] = dsHdr.GetDataStartFilePos();
00074                         rowsizes[ig][is] = dsHdr.GetRowSize();
00075             dataSetNames[ig][is] = dsHdr.GetName();
00076 
00077                         int nc = dsHdr.GetColumnCnt();
00078                         colsizes[ig][is].resize(nc);
00079                         for (int ic=0; ic<nc; ic++)
00080                                 colsizes[ig][is][ic] = dsHdr.GetColumnInfo(ic).GetSize();
00081                 }
00082         }
00083         reader.Close();
00084 }

00086 void DataSetUpdater::SeekToPosition(ofstream &os, int groupIndex, int setIndex, int row, int col)
00087 {
00088         int pos = positions[groupIndex][setIndex] + (row*rowsizes[groupIndex][setIndex]);
00089         for (int i=0; i<col; i++)
00090         {
00091                 pos += colsizes[groupIndex][setIndex][i];
00092         }
00093         os.seekp(pos, std::ios::beg);
00094 }
