00438 int32_t Util::schrageRandom(int32_t *ix) {
00439   int32_t a = 16807, p = 2147483647, xhi = 0, xlo = 0;
00440   int32_t rand = 0;
00441   int32_t r = p % a;
00442   int32_t q = p / a;
00443   Err::check(*ix > 0, "Error: Util::schrageRandom() - Cannot seed with 0");
00444   /* Break into high and low bits to avoid overflow. */
00445   xhi = (*ix) / q;
00446   xlo = (*ix)  - xhi*q;
00447   /* Combine high and low. */
00448   (*ix) = a * xlo - r * xhi;
00449   if((*ix) < 0) {
00450     (*ix) = (*ix) + p;
00451   }
00452   rand = (*ix);
00453   return rand;

00767 void Util::printStringWidth(std::ostream &out,const std::string& str,
00768                             int prefix,
00769                             int currentPos, int maxWidth ) {
00770   const char *wStart = NULL, *wEnd = NULL; /* Start and end of word pointers. */
00771   int position = currentPos;
00772   int nextSize = 0;
00773   int i = 0;
00774   wStart = str.c_str();
00775 
00776   /* While there are still characters to be printed. */
00777   while(*wStart != '\0') {
00778 
00779     /* Clean out any whitespace. */
00780     while(isspace(*wStart) && *wStart != '\0') {
00781       if(*wStart == '\n') {
00782         out.put('\n');
00783         for(i = 0; i < prefix; i++)
00784           out.put(' ');
00785         fflush(stdout);
00786         position = prefix;
00787       }
00788       *wStart++;
00789     }
00790 
00791     if(*wStart == '\0')
00792       break;
00793 
00794     /* Find the end of current word. */
00795     wEnd = wStart;
00796     while(!isspace(*wEnd) && *wEnd != '\0')
00797       wEnd++;
00798 
00799     /* Time for a newline? */
00800     if((wEnd - wStart) + position >= maxWidth) {
00801       out.put('\n');
00802       for(i = 0; i < prefix; i++)
00803          out.put(' ');
00804       position = prefix;
00805     }
00806 
00807     /* Print out the word. */
00808     while(wStart < wEnd) {
00809       out.put(*wStart);
00810       fflush(stdout);
00811       wStart++;
00812       position++;
00813     }
00814 
00815     /* Look to see where next word is. */
00816     while(isspace(*wEnd)) {
00817       if(*wEnd == '\n') {
00818         out.put('\n');
00819         for(i = 0; i < prefix; i++)
00820             out.put(' ');
00821         position = prefix;
00822       }
00823       wEnd++;
00824     }
00825 
00826     /* Figure out the size of the next word. */
00827     wStart = nextWhiteSpace(wEnd);
00828     if(wStart != NULL)
00829       nextSize = wStart - wEnd;
00830     else
00831       nextSize = 0;
00832 
00833     /* Print a space if we're not going to
00834        print a newline. */
00835     if(wEnd != '\0' &&
00836        nextSize + position < maxWidth &&
00837        position != 0) {
00838       out.put(' ');
00839       position++;
00840     }
00841 
00842     wStart = wEnd;
00843   }
00844 }
