01149 DataSetInfo *CHPMultiDataData::OpenMultiDataDataSet(MultiDataType dataType)
01150 {
01151         map<MultiDataType, DataSetInfo>::iterator pos = dataSetInfo.find(dataType);
01152         if (pos != dataSetInfo.end())
01153         {
01154                 return &dataSetInfo[dataType];
01155         }
01156         //read metrics column data
01157         DataSetInfo info;
01158     if (dataTypeGroupNames.empty() == true)
01159     {
01160         std::map<std::wstring, MultiDataType> nameTypeMap;
01161         int nnames = sizeof(MultiDataDataSetNames) / sizeof(std::wstring);
01162         for (int iname=0; iname<nnames; iname++)
01163             nameTypeMap[MultiDataDataSetNames[iname]] = MultiDataDataTypes[iname];
01164         int ng = genericData.Header().GetDataGroupCnt();
01165         for (int ig=0; ig<ng; ig++)
01166         {
01167             DataGroupHeader &dh = genericData.Header().GetDataGroup(ig);
01168             std::wstring name = dh.GetName();
01169             int ns = dh.GetDataSetCnt();
01170             for (int is=0; is<ns; is++)
01171             {
01172                 DataSetHeader &h = dh.GetDataSet(is);
01173                 dataTypeGroupNames[nameTypeMap[h.GetName()]] = dh.GetName();
01174             }
01175         }
01176     }
01177         info.entries = genericData.DataSet(dataTypeGroupNames[dataType], MultiDataDataSetNames[dataType]);
01178         if (info.entries)
01179         {
01180                 info.entries->Open();
01181                 int32_t ncols = info.entries->Header().GetColumnCnt();
01182                 info.metricColumns.clear();
01183                 int startCol = 0;
01184                 if (dataType == ExpressionMultiDataType || dataType == ExpressionControlMultiDataType)
01185                 {
01186                         startCol = 2;
01187                 }
01188                 else if (dataType == GenotypeMultiDataType || dataType == GenotypeControlMultiDataType)
01189                 {
01190                         startCol = 3;
01191                 }
01192                 else if (dataType == CopyNumberMultiDataType)
01193                 {
