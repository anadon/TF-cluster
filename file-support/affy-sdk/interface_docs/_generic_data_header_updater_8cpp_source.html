00068 bool GenericDataHeaderUpdater::UpdateFileId()

00071         if (currentHdr->GetFileId().length() == updateHdr->GetFileId().length() && updateHdr->GetFileId().length() != 0)

00073                 // Move the file stream position to the location of the file ID.
