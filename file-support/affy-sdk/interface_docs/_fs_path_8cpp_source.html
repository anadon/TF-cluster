00330 AptErr_t FsPath::setPath(const std::string& path) {
00331   std::string tmp_os_path;
00332   std::string tmp_internal_path;
00333   std::string::size_type pos;
00334 
00335   clear();
00336 
00337   // save it off the original path
00338   m_original_path=path;
00339 
00340   if (path=="") {
00341     return APT_OK;
00342   }
00343 
00344   // tmp gets chopped up as we work and path is const, so we need a copy.
00345   tmp_os_path=path;
00346 
00347   // convert any MS-like paths to normal paths.
00348   Fs::unixifyPathInPlace(tmp_os_path);
00349 
00350   // if there is a drive letter, pull it off first
00351   // note this is ambiguous with a single letter file name with an internal 
00352   // group/table
00353   // "c:foo"
00354   if (tmp_os_path[1]==':') {
00355     m_dir_drive=tmp_os_path.substr(0,2);
00356     tmp_os_path.erase(0,2);
00357   }
00358   
00359   // The last ":" of the string seperates the internal name
00360   // ...../FILE.EXT:INTERNAL_GROUP/INTERNAL_TABLE.
00361   pos=tmp_os_path.rfind(FSPATH_INTERNAL_PATHSEP);
00362   // order is important, pull out the internal ":group/table" first.
00363   if (pos!=std::string::npos) {
00364     tmp_internal_path=tmp_os_path.substr(pos+1);
00365     tmp_os_path.erase(pos,tmp_os_path.size());
00366   }
00367 
00368   // now we have two parts...
00369   // ...do the os part first...
00370   pos=tmp_os_path.rfind(FSPATH_PATHSEP);
00371   if (pos==std::string::npos) {
00372     // no dir parts.
00373     m_dir_name="";
00374     m_file_name=tmp_os_path;
00375   }
00376   else {
00377     m_dir_name=Fs::addTrailingSlash(tmp_os_path.substr(0,pos+1));
00378     if (pos<tmp_os_path.size()) {
00379       m_file_name=tmp_os_path.substr(pos+1);
00380     }
00381   }
00382 
00383   //dump();
00384 
00385   // try at ".EXT"
00386   pos=m_file_name.rfind(FSPATH_EXTSEP);
00387   if (pos!=std::string::npos) {
00388     m_file_ext=m_file_name.substr(pos+1);
00389     m_file_name.erase(pos);
00390   }
00391

00393   pos=tmp_internal_path.rfind(FSPATH_PATHSEP);

00395     m_internal_table_name=tmp_internal_path;
