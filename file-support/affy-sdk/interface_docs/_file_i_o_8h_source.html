00055 inline uint32_t
00056 affy_swap32(uint32_t x) {
00057   return ((((x) & 0xff000000) >> 24) | (((x) & 0x00ff0000) >>  8) |
00058           (((x) & 0x0000ff00) <<  8) | (((x) & 0x000000ff) << 24));
00059 }
00060 
00065 inline uint16_t
00066 affy_swap16(uint16_t x) {
00067   return ((((x) >> 8) & 0xff) | (((x) & 0xff) << 8));
00068 }
