00044 void SnpTable::open(const std::string& fileName)
00045 {
00046   affx::TsvFile tsv;
00047   std::string snpName;
00048   std::vector<int> gTypes;
00049   tsv.open(fileName);
00050   int colCount = tsv.getColumnCount(0);
00051   tsv.bind(0, "probeset_id", &snpName);
00052   gTypes.resize(colCount - 1);
00053   /* Bind up the variables. */
00054   for (int i = 1; i < colCount; i++) {
00055     std::string s;
00056     tsv.cidx2cname(0, i, s);
00057     m_ColNames.push_back(s);
00058     tsv.bind(0, i, &gTypes[i-1]);
00059     m_ColNameMap[s] = i - 1;
00060   }
00061   int count = 0;
00062   while (tsv.nextLevel(0) == TSV_OK) {
00063     m_RowNames.push_back(snpName);
00064     m_GTypes.push_back(gTypes);
00065     m_RowNameMap[snpName] = count++;
00066   }
00067   tsv.close();
00068 }
