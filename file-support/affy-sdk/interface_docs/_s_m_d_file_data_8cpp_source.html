00129 bool SMDFileData::ReadData()
00130 {
00131         Clear();

00133         CIniFile iniFile;
00134         std::vector<CIniFile::Record> allRecords = iniFile.GetAllRecords(filename);
00135

00137                 // Either the file does not exist or is not the correct format.

00139 
00140         int numFrames = 0;
00141         if (GetIntValue(allRecords, SummarySec, Frames, numFrames) == false)
00142                 return false;
00143 
00144         if (numFrames < 1)
00145                 return false;
00146 
00147         int numRows = 0;
00148         if (GetIntValue(allRecords, SummarySec, FrameRows, numRows) == false)
00149                 return false;
00150 
00151         int numCols = 0;
00152         if (GetIntValue(allRecords, SummarySec, FrameCols, numCols) == false)
00153                 return false;
00154 
00155         frameRows = numRows;
00156         frameCols = numCols;
00157 
00158         for (int iFrame = 0; iFrame < numFrames; ++iFrame)
00159         {
00160                 SMDFrame frame;
00161                 char sectionName[10];
00162                 snprintf(sectionName,sizeof(sectionName), "%d", iFrame);
00163 
00164                 frame.frameIdx = iFrame;
00165 
00166                 if (GetIntValue(allRecords, sectionName, KeyRows, frame.rows) == false)
00167                         return false;
00168                 if (GetIntValue(allRecords, sectionName, KeyCols, frame.cols) == false)
00169                         return false;
00170                 if (GetIntValue(allRecords, sectionName, KeyStartRow, frame.startRow) == false)
00171                         return false;
00172                 if (GetIntValue(allRecords, sectionName, KeyStartCol, frame.startCol) == false)
00173                         return false;
