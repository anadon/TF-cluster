00055 /*
00056  * Read the header and return the file type id. If the file fails to read
00057  * it must be an old CHP file, just return a blank for the file type id.
00058  */
00059 FusionCHPData *FusionCHPDataReg::CreateObject(const AffymetrixGuidType &id)
00060 {
00061         // Find the matching CHP data object.
00062         for (FusionCHPDataReg *p=m_Head; p != NULL; p = p->m_Next)
00063         {
00064                 AffymetrixGuidTypeList::iterator it;
00065                 for (it=p->fileTypeIdentifiers.begin(); it!=p->fileTypeIdentifiers.end(); ++it)
00066                 {
00067                         if (*it == id)
00068                         {
00069                                 FusionCHPData *chp = p->MakeObject();
00070                                 chp->fileTypeIdentifiers = p->fileTypeIdentifiers;
00071                                 chp->fileTypeIdentifier = id;
00072                                 if (chp)
00073                                 {
00074                                         return chp;
00075                                 }
00076                         }
00077                 }
00078         }
00079 
00080         // Go back now and find the generic reader (the one with no file type identifiers)
00081         for (FusionCHPDataReg *p=m_Head; p != NULL; p = p->m_Next)
00082         {
00083                 if (p->fileTypeIdentifiers.size() == 0)
00084                 {
00085                         FusionCHPData *chp = p->MakeObject();
00086                         chp->fileTypeIdentifier = id;
00087                         return chp;
00088                 }
00089         }
00090 
00091         return NULL;
00092 }
