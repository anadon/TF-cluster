00271   // do the diff in three passes.
00272 
00273   // 1 = changed values
00274   tsv1.headersBegin();

00276     if (tsv2.getHeader(key1,val2)==affx::TSV_OK) {
00277       if (val1==val2) {
00278         if (m_opt_print_same) {
00279           if (p_inc()) {
00280             printf(" #%%%s=%s\n",key1.c_str(),val1.c_str());

00282         }

00284       else { // !=

00287           printf("-#%%%s=%s\n",key1.c_str(),val1.c_str());

00289         }
00290       }
00291     }
00292   }
00293 
00294   // 2 = headers in tsv1 not in tsv2
00295   tsv1.headersBegin();
00296   while (tsv1.headersNext(key1,val1)==affx::TSV_OK) {
00297     if (tsv2.getHeader(key1,val2)!=affx::TSV_OK) {
00298       m_diff_cnt++;
00299       if (p_inc()) {
00300         printf("-#%%%s=%s\n",key1.c_str(),val1.c_str());
00301       }
00302     }
00303   }
00304   // 3 = headers in tsv2 not in tsv1
00305   tsv2.headersBegin();

00307     if (tsv1.getHeader(key2,val1)!=affx::TSV_OK) {
