00678 void PgOptions::matchOneArg(size_t* arg_idx)
00679 {
00680   PgOpt *opt;
00681   std::string arg_opt;
00682   std::string arg_val;
00683   int have_arg_val=0;
00684   size_t idx;
00685 
00686   if (!(*arg_idx<m_argv.size())) {
00687     Err::errAbort("out of bounds");
00688   }
00689 
00690   //printf(">> %3d : %s\n",*arg_idx,m_argv[*arg_idx].c_str()); // debug
00691   arg_opt=m_argv[*arg_idx];
00692   *arg_idx+=1;
00693 
00694   // No leading "-"; Add it to the args vec...
00695   if (arg_opt[0]!='-') {
00696     m_args.push_back(arg_opt);
00697     return;
00698   }
00699 
00700   // "--" terminates options; Everything following "--" is an args.
00701   if (arg_opt=="--") {
00702     while (*arg_idx<m_argv.size()) {
00703       m_args.push_back(m_argv[*arg_idx]);
00704       *arg_idx+=1;
00705     }
00706     return;
00707   }
00708 
00709   // strip off "-" or "--"
00710   if (arg_opt[0]=='-') {
00711     if (arg_opt[1]=='-') {
00712       arg_opt.erase(0,2); // "--"
00713     }
00714     else {
00715       arg_opt.erase(0,1); // "-"
00716     }
00717   }
00718   else { // no "-" shouldnt happen.
00719     Err::errAbort("Argument does not have a starting dash: '"+arg_opt+"'");
00720   }
00721 
00722   // split on "=" for 'opt=val'...
00723   idx=arg_opt.find("=");
00724   if (idx!=std::string::npos) {
00725     have_arg_val=1;
00726     arg_val.assign(arg_opt,idx+1,arg_opt.size()-(idx+1));
00727     arg_opt.erase(idx); // erase "=..."
00728   }
00729 
00730   // should have something left
00731   if (arg_opt=="") {
00732     Err::errAbort("Shouldnt have a blank argument.");
00733   }
00734 
00735   // Might this be a negated arg? (like "--no-print"?)
00736   if (arg_opt.find("no-")==0) {
00737     string arg_opt_no=arg_opt;
00738     arg_opt_no.erase(0,3);
00739     opt=findOpt(arg_opt_no);
00740     if (opt!=NULL) {
00741       if (opt->m_type!=PgOpt::BOOL_OPT) {
00742         Err::errAbort("Cant use '--no-' with '"+arg_opt_no+"': Not a boolean option.");
00743       }
00744       opt->setValue("false");
00745       return;
00746     }
00747     // didnt find it, treat it as a normal option.
00748   }
00749 
00750   opt = findOpt(arg_opt);
00751   if(opt == NULL) {
00752     Err::errAbort("Don't recognize option: '" + arg_opt + "'");
00753   }
00754 
00755 //  if (*arg_idx<m_argv.size()) {
00756 //    printf("   %3d : %s\n",*arg_idx,m_argv[*arg_idx].c_str()); // debug
00757 //  }
00758 
00759   if (opt->m_type == PgOpt::BOOL_OPT) {
00760     // --foo=bar
00761     if (have_arg_val==1) {
00762       opt->checkParseIsOk(arg_val);
00763       opt->setValue(arg_val);
00764     }
00765     // we might have a following "true/false"...
00766     // "--foo true"
00767     else {
00768       if (*arg_idx<m_argv.size()) {
00769         bool success;
00770         // printf("test: m_argv[%d] = '%s'\n",*arg_idx,m_argv[*arg_idx].c_str()); // debug
00771         Convert::toBoolCheck(m_argv[*arg_idx], &success);
00772         if (success) {
00773           opt->setValue(m_argv[*arg_idx]);
00774           *arg_idx+=1;
00775         }
00776         else {
00777           // didnt look like a bool, use 
00778                         opt->setValue("true");
00779         }
00780       }
00781       // just "program --foo"
00782       else {
00783         opt->setValue("true");
00784       }
00785     }
00786     //
00787     return;
00788   }
00789 
00790   // Do we take the next value?
00791   if (have_arg_val==0) {
00792     if (*arg_idx<m_argv.size()) {
00793       arg_val=m_argv[*arg_idx];
00794       *arg_idx+=1;
00795     }
00796     else {
00797       arg_val="";
00798     }
00799   }
00800 
00801   // Add the value
00802   if (opt->checkParseIsOk(arg_val)==1) {
00803     if (opt->m_allowMultiple==1) {
00804       opt->pushValue(arg_val);
00805     }
00806     else {
00807           if (opt->m_longName == "xml-file") {
00808                 if (opt->isSet()) {Err::errAbort("The xml-file option has already been set. Only one xml-file can be specified.");}
00809                 std::vector<std::string> vFileNames;
00810                 setOptionsFromXMLFile(arg_val, vFileNames);
00811           }
00812       opt->setValue(arg_val);
00813     }
00814   }
00815   else {
00816     Err::errAbort("bad parse setting '" + arg_opt +"' to '" + arg_val + "'");
00817   }
00818 }

00865         {
00866                 if (Fs::basename(strFileName) == vFileNames[iIndex]) {Err::errAbort("Possible run away recursion situation found in PgOptions::setOptionsFromXMLFile(...)");}
00867         }

00870         if (m_strXMLParameterFileName == "")
00871         {
00872                 m_strXMLParameterFileName = strFileName;
00873         }
00874         // Initialize the XML4C system
00875         try

00878         }
00879 
00880         catch (const XERCES_CPP_NAMESPACE::XMLException& toCatch)
00881         {                                       
00882                 Err::errAbort("PgOptions::setParametersFromXMLFile() failed at XMLPlatformUtils::Initialize(). Msg: " + PgOptionsSAXHandler::toString(toCatch.getMessage()) + " FileName: " + strFileName);
00883         }
00884         
00885         //
00886         //  Create a SAX parser object to use and create our SAX event handlers
00887         //  and plug them in.
00888         //
00889         XERCES_CPP_NAMESPACE::SAXParser* parser = new XERCES_CPP_NAMESPACE::SAXParser;
00890         PgOptionsSAXHandler handler(this, vFileNames);
00891         parser->setDocumentHandler(&handler);
00892         parser->setErrorHandler(&handler);

00906         {

00908                 XERCES_CPP_NAMESPACE::XMLPScanToken token;

00910                 try
00911                 {

00914                         {

00918                                         Err::errAbort("PgOptions::setParametersFromXMLFile() Cannot open or parse xml-file. FileName: " + strFileName);

00928                         }

00931                         //  We started ok, so lets call scanNext() until we find what we want
00932                         //  or hit the end.
00933                         //

00935                         while (gotMore && !parser->getErrorCount())

00937                                 gotMore = parser->parseNext(token);
00938                         }
