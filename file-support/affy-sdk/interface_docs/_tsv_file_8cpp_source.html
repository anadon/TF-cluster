01118                                      std::multimap<T1,linenum_t>& mmap,

01129     r_start=mmap.begin();

01132   else if (op==TSV_OP_EQ) {
01133     r_start=mmap.lower_bound(val);

02919 int
02920 affx::TsvFile::f_read_column(TsvFileField* col)
02921 {
02922   int c;
02923   //char* buf_ptr;
02924   //int   buf_len;
02925   int   bi;
02926   char  in_quotes;
02927 
02928   //
02929   assert(col!=NULL);
02930   // mark this column as being null.
02931   col->m_isnull=true;
02932   col->m_value_int_done=false;
02933   col->m_value_double_done=false;
02934   col->m_value_uint_done=false;
02935   col->m_value_ulonglong_done=false;
02936 
02937   // Nothing to be read.
02938   if (!m_fileStream.good()) {
02939     col->m_buffer="";
02940     return (TSV_ERR_FILEIO);
02941   }
02942 
02943   // Mark it as not being null.
02944   col->m_isnull=false;
02945 
02946   in_quotes=0;
02947 
02948   // The common case is the field is the same size as last time.
02949   // Copy the chars into an already existing string.
02950   bi=0;
02951   col->m_buffer.resize(0);
02952   while (1) {
02953     //c=m_fileStream.get();
02954     c=M_GETC();
02955 
02956     //
02957     if ((m_optEscapeChar==c)&&(m_optEscapeOk==true)) {
02958       //c=m_fileStream.get();
02959       c=M_GETC();
02960       c=unescapeChar(c);
02961     }
02962     // start or end of quotes?
02963     else if (((c==m_optQuoteChar1)&&(m_optQuoteChar1!=0))||
02964              ((c==m_optQuoteChar2)&&(m_optQuoteChar2!=0)))
02965       {
02966         if (in_quotes==0) { // open
02967           in_quotes=c;
02968         }
02969         else if (c==in_quotes) { // close
02970           in_quotes=0;
02971         }
02972       }
02973     else if ((m_optFieldSep==c)&&(in_quotes==0)) {
02974       col->m_buffer.resize(bi); // trunc the buffer to size.
02975       break; // discard the field-sep-char
02976     }
02977     // end of field? (short value)
02978     else if ((c==TSV_CHAR_CR)||(c==TSV_CHAR_LF)) {
02979       //m_fileStream.putback(c); // leave eol char
02980       M_UNGETC(c);
02981       col->m_buffer.resize(bi); // trunc
02982       break;
02983     }
02984     // ran out of data
02985     else if (c==EOF) {
02986       if (bi==0) {
02987         col->m_isnull=true; // revise our opinion about this being non-null
02988       }
02989       col->m_buffer.resize(bi);
02990       return (TSV_ERR_FILEIO);
02991     }
02992 
02993     // add to our buffer
02994     col->m_buffer.append(1,(char)c); // buffer full, append
02995     bi++;
02996   }
02997 
02998   // dequote the column?
02999   if (col->m_optAutoDequote==true) {
03000     trim(col->m_buffer);
03001     dequote(col->m_buffer);
03002   }
03003 
03004   //printf("z\n"); fflush(NULL);
03005   return TSV_OK;
03006 }

03337 int
03338 affx::TsvFile::nextLevel(int seek_clvl)
03339 {
03340   char c = '\0';
03341   // insure the vars are inplace
03342   // linkvars_maybe();
03343 
03344   // Read lines until we find a line of the correct level.
03345   while (1) {
03346     if (!m_fileStream.good()) {
03347       clearFields();
03348       m_eof=true;
03349       return (TSV_ERR_FILEIO);
03350     }
03351     c = M_PEEK();
03352     // speed tweak to avoid uncessary seek. Unless this is the 0 level
03353     // the first character should be a tab. This handles a common case
03354     // in pgf files.
03355     if(seek_clvl != 0 && c != TSV_CHAR_TAB && c != TSV_CHAR_COMMENT) {
03356       return TSV_LEVEL_LAST;
03357     }
03358     else if (c == EOF) {
03359       clearFields();
03360       return TSV_ERR_EOF;
03361     }
03362 
03363     // what level is this line at?
03364     m_lineLvl=f_advance_tabs();
03365 
03366     // skip the line if it isnt a data line.
03367     if (m_lineLvl<0) {
03368       f_advance_eol();
03369       continue;
03370     }
03371 
03372     // a match; bind and return
03373     if (m_lineLvl==seek_clvl) {
03374       return f_read_columns(m_lineLvl);
03375     }
03376     // a master; dont skip it
03377     if (m_lineLvl<seek_clvl) {
03378       m_fileStream.seekg(m_line_fpos);
03379       return TSV_LEVEL_LAST;
03380     }
03381     // a child; skip it
03382     if (m_lineLvl>seek_clvl) {
03383       f_advance_eol();
03384     }
03385   }
03386   return (TSV_ERR_UNKNOWN);
03387 }

03397 {

03400   int clvl_max;

03446       seen_all_levels=1;

03449   }
03450

04051 {
04052   vector<linenum_t>::iterator lb;
04053   linenum_t sline;
04054 
04055   // do some bookkeeping first

04057   indexBuildMaybe();

04064     }

04070   for (unsigned int clvl=0;clvl<m_index_lineclvl.size();clvl++) {

04080     seekLine(sline);
04081     nextLine();

04083     if (sline==line) {
04084       break;
04085     }
04086   }

04830   // skip non key=val headers

04832     iter++;
04833   }
04834   if (iter==m_headers_bykey.end()) {
04835     return 0;
04836   }
04837   // walk the keys in order looking for dups
04838   while (1) {
04839     iter_last=iter;
04840     iter++;
04841     // done?
04842     if (iter==m_headers_bykey.end()) {
04843       break;
04844     }
04845     // a dup?
04846     if (iter->second->m_key==iter_last->second->m_key) {
04847       // print all the dup headers
04848       iter_last->second->print();
04849       dup_cnt++;
04850       while (iter->second->m_key==iter_last->second->m_key) {
04851         iter->second->print();
04852         dup_cnt++;
04853         iter++;
04854       }
04855     }
04856   }
04857   //
04858   return dup_cnt;
