01184 /*
01185  * Get the chip type (probe array type) of the CDF file.
01186  * This is the name of the file without extension for CDF XDA format version < 4. We
01187  * also include all substrings create by removing
01188  * characters to the right of each '.'
01189  * For CDF XDA format version >= 4, this is retrieved from the file header.
01190  */
01191 std::vector<std::string> FusionCDFData::GetChipTypes()
01192 {
01193         if (!gcosData && !calvinData)
01194                 CreateObject();
01195         if (gcosData)
01196         {
01197                 gcosData->SetFileName(fileName.c_str());
01198                 return gcosData->GetChipTypes();
01199         }
01200         else
01201         {
01202                 std::vector<std::string> chiptypes;
01203                 std::string chiptype;
01204 
01205                 int index = (int) fileName.rfind('\\');
01206                 if (index == -1)
01207                         index = (int) fileName.rfind('/');
01208                 chiptype = fileName.c_str() + index + 1;
01209                 chiptype.resize(chiptype.length()-4);
01210 
01211                 // The full file name (minus .cdf extension) is the default (1st) 
01212                 // chip type. This matches what GetChipType() returns.
01213                 // ie: foo.bar.v1.r2.cdf -> foo.bar.v1.r2
01214                 chiptypes.push_back(chiptype);
01215 
01216                 //We then add all substrings starting at zero and ending at '.'
01217                 // ie: foo.bar.v1.r2.cdf -> foo.bar.v1, foo.bar, foo
01218                 std::string::size_type pos = chiptype.rfind(".",chiptype.size()-1);
01219                 while (pos != std::string::npos){
01220                         if(pos>0)
01221                                 chiptypes.push_back(chiptype.substr(0,pos));
01222                         pos = chiptype.rfind(".",pos-1);
01223                 }
01224 
01225                 //ie: foo.bar.v1.r2, foo.bar.v1, foo.bar, foo
01226                 return chiptypes;
01227         }
01228 }
