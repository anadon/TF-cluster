00582       m_max_code = candidate_max;      
00583       max_cn++;
00584       int bound = max_cn;
00585       if (bound > effective_max_allele_count) {
00586         bound = effective_max_allele_count;
00587       }
00588       for (int allele_idx = 1; allele_idx <= bound; allele_idx++) {
00589         candidate_max += choose(effective_max_allele_count, allele_idx) * 
00590           choose(max_cn - 1, allele_idx - 1);
00591       }
00592     }
00593     max_cn--;
00594 
00595     // fill out m_abstract_codes[] with abstract alleles.  
00596 
00597     // this is a bootstrap method.  allele strings for each copy number are
00598     // built upon the allele strings for the previous copy number.  First the
00599     // array is initialized with CN=1 alleles: A, B, C, ... n.  CN=2 alleles
00600     // are generated by iteratively prepending single allele letters to the
00601     // previous CN=1 alleles.  E.g. 'A' is prepended to all CN=1 alleles
00602     // starting with the A allele to make AA, AB, AC, ... An, and 'B' is
00603     // prepended to all of the alleles starting with the B allele: BB, BC, ...,
00604     // Bn, and 'C' is prepended to all the alleles starting with the C allele,
00605     // and so on.  The hard part is in the index bookkeeping.  For each
00606     // prepending allele letter ('A','B', etc.), an index is kept of where in
00607     // the array that letter needs to start prepending (start_idxs[]).  An
00608     // index is kept of where to stop prepending (tail_stop_idx).  After each
00609     // allele letter completes a series of prepending, it's start index is
00610     // updated so that it will begin prepending in the correct place for the
00611     // next copynumber.
00612 
00613     // initialize for haploid alleles
00614     int m_abstract_codes_end = m_min_translatable_code;
00615 
00616     vector<int> start_idxs(effective_max_allele_count, m_min_translatable_code - 1);
00617     vector<string> prepend_alleles(effective_max_allele_count);
00618     int i;
00619     for (i = 0; i < max_allele_count; i++) {
00620       prepend_alleles[i] = string(1, m_num2alpha[i]);
00621     }
00622 
00623     prepend_alleles[i] = 'n';
00624 
00625     for (int cn = 1; cn <= max_cn; cn++) {
00626       int tail_stop_idx = m_abstract_codes_end;
00627       for (i = 0; i < start_idxs.size(); i++) {
00628         // save the start_idx for the next copy number 
00629         int next_start_idx = m_abstract_codes_end;
00630         for (int curr_idx = start_idxs[i]; curr_idx < tail_stop_idx; curr_idx++) {
00631           m_abstract_codes[m_abstract_codes_end++] = prepend_alleles[i] + m_abstract_codes[curr_idx];
00632         }
00633         start_idxs[i] = next_start_idx;
00634       }
00635     }
00636 
00637     // add non-translatable codes
00638     m_abstract_codes[m_min_code] = "NoCall";  // Unable to genotype
00639     m_abstract_codes[m_min_code + 1] = "ZeroCopyNumber";
00640     //m_abstract_codes[17] = "0";
00641 
00642     // add extra codes for this version
00643     for (i = 0; i < extra_codes_count; i++) {
00644       m_abstract_codes[m_max_extra_code - i] = m_extra_codes[i];
00645     }
00646 
00647     // insert allele delimiter if one was provided
00648     if (allele_delimiter != '\0') {
00649       for (i = m_min_translatable_code; i <= m_max_code; i++) {
00650         string new_allele_string;
00651         new_allele_string = m_abstract_codes[i][0];
00652         for (size_t j = 1; j < m_abstract_codes[i].size(); j++) {
00653           new_allele_string += allele_delimiter;
00654           new_allele_string += m_abstract_codes[i][j];
00655         }
00656         m_abstract_codes[i].assign(new_allele_string);
00657       }
00658     }
00659   }
00660   // make reverse map of abstract allele codes for encoding 
00661   for (i = m_min_code; i <= m_max_code; i++) {
00662     m_alleles[m_abstract_codes[i]] = i;
00663   }
00664   for (i = m_min_extra_code; i <= m_max_extra_code; i++) {
00665     m_alleles[m_abstract_codes[i]] = i;
00666   }
00667   
00668   m_allele_delimiter = allele_delimiter;
00669 }
