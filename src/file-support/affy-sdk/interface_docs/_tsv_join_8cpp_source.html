00369 void tsvJoin::writeMatches (void)
00370 {
00371   Verbose::out (1, "Done indexing files. Generating merged file.");
00372 
00373   // Write column names only if find at least one match.
00374   bool columnNamesWritten = false;
00375 
00376   // Read stream file.
00377   while (m_StreamTsv.nextLevel (0) == TSV_OK)
00378   {
00379     bool foundMatch = false;
00380     // Look for matches in each index file.
00381     for (unsigned int idx = 0; idx < m_FileCount - 1; ++idx)
00382     {
00383       TsvFile* indexTsv = m_IndexTsvs[idx];
00384       if (indexTsv->findBegin (0, m_Key, TSV_OP_EQ, m_StreamKeyValue) != TSV_OK)
00385         Err::errAbort ("Problem reading index file " + m_FileNames[idx + 1]);
00386 
00387       const int resultCount = indexTsv->findResultsCount();
00388       // Require a match in all index files.
00389       if (resultCount == 0)
00390       {
00391         foundMatch = false;
00392         break;
00393       }
00394       // Fatal error if more than one match was found.
00395       else if (resultCount > 1)
00396       {
00397         string msg = "FATAL: key '" + m_Key + "' is not a unique index. Duplicate key found, [";
00398         msg += m_StreamKeyValue + "] for file " + m_FileNames[idx + 1] + ".";
00399         Err::errAbort (msg);
00400       }
00401       // Found one match - read data
00402       if (indexTsv->findNext() != TSV_OK)
00403         Err::errAbort ("Problem reading index file " + m_FileNames[idx + 1]);
00404       foundMatch = true;
00405     } // end for (unsigned int idx = 0; idx < m_FileCount - 1; ++idx)
00406 
00407     if (foundMatch)
00408     {
00409       // Found match - write column names once.
00410       if (! columnNamesWritten)

00414       }
00415       // Write key.
