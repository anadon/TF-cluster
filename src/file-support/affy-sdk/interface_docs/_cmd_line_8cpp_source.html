00152 bool CCmdLine::IsSwitch(const char *pParam)
00153 {
00154    if (pParam==NULL)
00155       return false;
00156 
00157    // switches must non-empty
00158    // must have at least one character after the '-'
00159    int len = (int) strlen(pParam);
00160    if (len <= 1)
00161    {
00162       return false;
00163    }
00164 
00165    // switches always start with '-'
00166    if (pParam[0]=='-')
00167    {
00168       // allow negative numbers as arguments.
00169       // ie., don't count them as switches
00170       return (!isdigit(pParam[1]));
00171    }
00172    else
00173    {
00174       return false;
00175    }
